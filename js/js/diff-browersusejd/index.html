Js识别不同的浏览器

2023/2/26 js js

Navigator 对象包含有关浏览器的信息。我们可以使用Navigator 对象的userAgent 属性来获取浏览器的信息。

userAgent 属性是一个只读的字符串，声明了浏览器用于 HTTP 请求的用户代理头的值。

1.判断浏览器，不判断版本

   const userAgent = navigator.userAgent; // 获取浏览器信息
        console.log(userAgent)
        if (userAgent.includes('Firefox')) {
            // Firefox浏览器
            // 执行Firefox浏览器特定的代码
            console.log('Firefox')
        } else if (userAgent.includes('Chrome')) {
            console.log('Chrome')
            // Chrome浏览器
            // 执行Chrome浏览器特定的代码
        } else if (userAgent.includes('Safari')) {
            console.log('Safari')
            // Safari浏览器
            // 执行Safari浏览器特定的代码
        } else if (userAgent.includes('Opera') || userAgent.includes('OPR')) {
            console.log('Opera')
            // Opera浏览器
            // 执行Opera浏览器特定的代码
        } else if (userAgent.includes('Edge')) {
            console.log('Edge')
            // Edge浏览器
            // 执行Edge浏览器特定的代码
        } else {
            console.log("其他浏览器")
            // 其他浏览器
            // 执行其他浏览器特定的代码
        }
  1. 判断浏览器，并判断版本
    const userAgent = navigator.userAgent.toLowerCase();
    console.log(userAgent)
    const isIE = userAgent.includes('msie') || userAgent.includes('trident');
    const isFirefox = userAgent.includes('firefox');
    const isChrome = userAgent.includes('chrome') && !userAgent.includes('edge');
    const isSafari = userAgent.includes('safari') && !userAgent.includes('chrome');
    const isOpera = userAgent.includes('opr') || userAgent.includes('opera');
    const isEdge = userAgent.includes('edge');
    if (isIE) {
      const ieVersion = parseInt(userAgent.match(/(?:msie |rv:)(\d+(\.\d+)?)/)[1]);
      // 执行IE浏览器特定的代码
      console.log(ieVersion,"IE");
    } else if (isFirefox) {
      const firefoxVersion = parseInt(userAgent.match(/firefox\/(\d+(\.\d+)?)/)[1]);
      // 执行Firefox浏览器特定的代码
      console.log(firefoxVersion,"Firefox");
    } else if (isChrome) {
      const chromeVersion = parseInt(userAgent.match(/chrome\/(\d+(\.\d+)?)/)[1]);
      // 执行Chrome浏览器特定的代码
      console.log(chromeVersion,"chrome");
    } else if (isSafari) {
      const safariVersion = parseInt(userAgent.match(/version\/(\d+(\.\d+)?)/)[1]);
      // 执行Safari浏览器特定的代码
      console.log(safariVersion,"safari");
    } else if (isOpera) {
      const operaVersion = parseInt(userAgent.match(/(?:opr|opera)[\s/](\d+(\.\d+)?)/)[1]);
      // 执行Opera浏览器特定的代码
      console.log(operaVersion,"opera");
    } else if (isEdge) {
      const edgeVersion = parseInt(userAgent.match(/edge\/(\d+(\.\d+)?)/)[1]);
      // 执行Edge浏览器特定的代码
      console.log(edgeVersion,"edge");
    } else {
      // 其他浏览器
      // 执行其他浏览器特定的代码
      console.log('其他浏览器');
    }
